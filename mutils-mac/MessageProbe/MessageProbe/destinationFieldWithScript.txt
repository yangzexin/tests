
tell application "Messages"
    activate
end tell

tell application "System Events"
	tell process "Messages"
        tell application "Messages"
            activate
        end tell
		set elements to get entire contents of front window
		repeat with i from 1 to length of elements
			set element to item i of elements
			if class of element as string = "text field" then
				set tmpTitle to title of element as string
				if tmpTitle is not "" then
					set positionInfo to position of element as string
					set cmd to "001"
					set the clipboard to cmd & positionInfo
                    
					tell application "MessageProbe"
						activate
					end tell

					tell process "MessageProbe"
						keystroke "i" using {command down}
					end tell
					exit repeat
				end if
			end if
		end repeat
	end tell
end tell
